\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $p(k) = \frac{1}{k+3}$. Compute

\[
\lim_{ h\to 0 } \frac{p(-1+h)-p(-1)}{h} \begin{prompt}=\answer{-\frac{1}{4}}\end{prompt}
\]
